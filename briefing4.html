function startBriefing(keyword){

  currentKeyword=keyword;

  document.getElementById('home').style.display='none';

  const screen=document.getElementById('briefing-screen');

  screen.style.display='flex';

  screen.className='active theme-'+keyword.toLowerCase();

  document.getElementById('audio-status').textContent='loading';

  document.getElementById('progress-bar').style.width='0%';

  stopSpeech();

}

}

async function fetchBriefing(keyword){

  try{

    const response=await fetch('https://tight-truth-449c.mecca2425.workers.dev/',{

      method:'POST',

      headers:{

        'Content-Type':'application/json',

        

        

        

      },

      body:JSON.stringify({

        model:'claude-sonnet-4-20250514',

        max_tokens:1000,

        tools:[{type:'web_search_20250305',name:'web_search'}],

        messages:[{role:'user',content:PROMPTS[keyword]+'\n\nToday is '+now.toDateString()+'. Search for content from the last 7 days only.'}]

      })

    });

    const data=await response.json();

    let text='';

    if(data.content){for(const block of data.content){if(block.type==='text')text+=block.text;}}

    if(!text)throw new Error(JSON.stringify(data));

    renderBriefing(text);

  }catch(err){

    document.getElementById('briefing-body').innerHTML=`<div class="error-msg">Couldn't load briefing.<br>Check connection and try again.<br><br><small>${err.message}</small></div>`;

    document.getElementById('audio-status').textContent='error';

  }

}

function renderBriefing(text){

  fullText=text;

  const lines=text.split('\n').filter(l=>l.trim());

  const items=[];let current='';

  for(const line of lines){

    if(/^\d+[\.\)]/.test(line.trim())){if(current)items.push(current.trim());current=line.replace(/^\d+[\.\)]\s*/,'');}

    else if(current){current+=' '+line.trim();}

    else{current=line.trim();}

  }

  if(current)items.push(current.trim());

  if(items.length===0)items.push(text);

  const html=items.map((item,i)=>`<div class="briefing-item"><div class="item-stripe"></div><div class="item-number">0${i+1}</div><div class="item-text">${item}</div></div>`).join('');

  document.getElementById('briefing-body').innerHTML=`<div class="briefing-items">${html}</div>`;

  document.getElementById('audio-status').textContent='tap play to listen';

  setTimeout(()=>{

    const els=document.querySelectorAll('.item-text');

    const spokenText=Array.from(els).map(el=>el.textContent).join('. ');

    autoPlay(spokenText);

  },800);

}

function autoPlay(text){if(window.speechSynthesis)speak(text);}

function speak(text){

  stopSpeech();

  if(!window.speechSynthesis){document.getElementById('audio-status').textContent='no tts available';return;}

  utterance=new SpeechSynthesisUtterance(text);

  utterance.rate=0.92;utterance.pitch=1.0;utterance.volume=1.0;

  const voices=speechSynthesis.getVoices();

  const preferred=voices.find(v=>v.name.includes('Samantha')||v.name.includes('Karen')||v.name.includes('Moira')||v.name.includes('Google UK English Female'));

  if(preferred)utterance.voice=preferred;

  utterance.onstart=()=>{isSpeaking=true;isPaused=false;document.getElementById('play-btn').textContent='⏸';document.getElementById('audio-status').textContent='playing';startProgress();};

  utterance.onend=()=>{isSpeaking=false;isPaused=false;document.getElementById('play-btn').textContent='▶';document.getElementById('audio-status').textContent='done';document.getElementById('progress-bar').style.width='100%';stopProgress();};

  utterance.onerror=()=>{isSpeaking=false;document.getElementById('play-btn').textContent='▶';document.getElementById('audio-status').textContent='tap play to listen';stopProgress();};

  speechSynthesis.speak(utterance);

}

function togglePlay(){

  if(!window.speechSynthesis)return;

  if(isSpeaking&&!isPaused){speechSynthesis.pause();isPaused=true;isSpeaking=false;document.getElementById('play-btn').textContent='▶';document.getElementById('audio-status').textContent='paused';stopProgress();}

  else if(isPaused){speechSynthesis.resume();isPaused=false;isSpeaking=true;document.getElementById('play-btn').textContent='⏸';document.getElementById('audio-status').textContent='playing';startProgress();}

  else if(fullText){const els=document.querySelectorAll('.item-text');const text=Array.from(els).map(el=>el.textContent).join('. ');speak(text||fullText);}

}

function stopSpeech(){

  if(window.speechSynthesis)speechSynthesis.cancel();

  isSpeaking=false;isPaused=false;

  document.getElementById('play-btn').textContent='▶';

  document.getElementById('audio-status').textContent='ready';

  document.getElementById('progress-bar').style.width='0%';

  stopProgress();

}

let progressStart=0,progressDuration=0;

function startProgress(){

  stopProgress();progressStart=Date.now();

  const wordCount=(fullText||'').split(/\s+/).length;

  progressDuration=(wordCount/130)*60*1000;

  progressInterval=setInterval(()=>{const elapsed=Date.now()-progressStart;const pct=Math.min((elapsed/progressDuration)*100,98);document.getElementById('progress-bar').style.width=pct+'%';},300);

}

function stopProgress(){if(progressInterval){clearInterval(progressInterval);progressInterval=null;}}

if(window.speechSynthesis){speechSynthesis.getVoices();speechSynthesis.onvoiceschanged=()=>speechSynthesis.getVoices();}
